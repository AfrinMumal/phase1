STEP 1E — Executive Dashboard (Final Consolidated Query)

SELECT 

    -- Total Customers & Loans
    COUNT(DISTINCT customer_id) AS total_customers,
    COUNT(*) AS total_loans,

    -- Status Counts
    COUNT(CASE WHEN loan_status = 'Active' THEN 1 END) AS active_loans,
    COUNT(CASE WHEN loan_status = 'Closed' THEN 1 END) AS closed_loans,
    COUNT(CASE WHEN loan_status = 'Defaulted' THEN 1 END) AS defaulted_loans,
    COUNT(CASE WHEN loan_status = 'Overdue' THEN 1 END) AS overdue_loans,

    -- Percentages
    ROUND(COUNT(CASE WHEN loan_status = 'Defaulted' THEN 1 END)*100.0/COUNT(*),2) AS default_rate_pct,
    ROUND(COUNT(CASE WHEN loan_status IN ('Defaulted','Overdue') THEN 1 END)*100.0/COUNT(*),2) AS portfolio_at_risk_pct,

    -- Financials
    ROUND(SUM(loan_amount)/10000000,2) AS total_portfolio_cr,
    ROUND(AVG(loan_amount)/100000,2) AS avg_loan_size_lakh,

    ROUND(SUM(CASE WHEN loan_status = 'Defaulted' THEN loan_amount END)/10000000,2) AS defaulted_amount_cr,
    ROUND(SUM(CASE WHEN loan_status IN ('Defaulted','Overdue') THEN loan_amount END)/10000000,2) AS at_risk_amount_cr,

    -- Portfolio Health Status (Combined)
    CASE 
        WHEN COUNT(CASE WHEN loan_status = 'Defaulted' THEN 1 END)*1.0/COUNT(*) <= 0.05 
            THEN 'HEALTHY - Standard monitoring protocols'
        
        WHEN COUNT(CASE WHEN loan_status = 'Defaulted' THEN 1 END)*1.0/COUNT(*) <= 0.10 
            THEN 'MODERATE RISK - Enhanced monitoring required'
        
        WHEN COUNT(CASE WHEN loan_status = 'Defaulted' THEN 1 END)*1.0/COUNT(*) <= 0.15 
            THEN 'HIGH RISK - Enhanced collection + review underwriting'
        
        ELSE 
            'CRITICAL - Crisis team + external recovery agencies'
    END AS portfolio_health_status

FROM edufin_small.loans;
